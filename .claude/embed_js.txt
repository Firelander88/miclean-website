// Use rootMargin so elements trigger slightly before entering viewport
var rvo = new IntersectionObserver(function(entries) {
  entries.forEach(function(x) {
    if (x.isIntersecting) { x.target.classList.add('in'); rvo.unobserve(x.target); }
  });
}, { threshold: 0.04, rootMargin: '0px 0px 60px 0px' });
document.querySelectorAll('.rv').forEach(function(el) { rvo.observe(el); });

// Staggered fade-in for cards/items — fallback makes everything visible after 2s
var co = new IntersectionObserver(function(entries) {
  entries.forEach(function(x) {
    if (x.isIntersecting) {
      x.target.style.opacity = '1';
      x.target.style.transform = 'none';
      co.unobserve(x.target);
    }
  });
}, { threshold: 0.03, rootMargin: '0px 0px 80px 0px' });

// ── Modal helpers ────────────────────────────────────────────────────────────
var _prevFocus = null;
function openM(n) {
  _prevFocus = document.activeElement;
  var overlay = document.getElementById('modal-' + n);
  overlay.classList.add('open');
  overlay.setAttribute('aria-hidden', 'false');
  document.body.style.overflow = 'hidden';
  if (n === 'quote') populateQuoteBasket();
  // Move focus to first focusable element inside modal
  setTimeout(function() {
    var first = overlay.querySelector('input,select,textarea,button,[tabindex]:not([tabindex="-1"])');
    if (first) first.focus();
  }, 50);
}
function closeM(n) {
  var overlay = document.getElementById('modal-' + n);
  overlay.classList.remove('open');
  overlay.setAttribute('aria-hidden', 'true');
  document.body.style.overflow = '';
  if (_prevFocus) { _prevFocus.focus(); _prevFocus = null; }
}
document.querySelectorAll('.modal-overlay').forEach(function(m) {
  m.setAttribute('role', 'dialog');
  m.setAttribute('aria-modal', 'true');
  m.setAttribute('aria-hidden', 'true');
  m.addEventListener('click', function(e) { if (e.target === m) { m.classList.remove('open'); m.setAttribute('aria-hidden','true'); document.body.style.overflow = ''; } });
  // Focus trap: keep Tab/Shift+Tab inside modal
  m.addEventListener('keydown', function(e) {
    if (!m.classList.contains('open')) return;
    if (e.key !== 'Tab') return;
    var focusable = Array.from(m.querySelectorAll('input,select,textarea,button,a[href],[tabindex]:not([tabindex="-1"])')).filter(function(el) { return !el.disabled; });
    if (!focusable.length) return;
    var first = focusable[0], last = focusable[focusable.length - 1];
    if (e.shiftKey) { if (document.activeElement === first) { e.preventDefault(); last.focus(); } }
    else { if (document.activeElement === last) { e.preventDefault(); first.focus(); } }
  });
});
document.addEventListener('keydown', function(e) {
  var tag = document.activeElement ? document.activeElement.tagName : '';
  var inInput = (tag === 'INPUT' || tag === 'TEXTAREA' || tag === 'SELECT');

  // Esc — close any open modal
  if (e.key === 'Escape') {
    document.querySelectorAll('.modal-overlay.open').forEach(function(m) {
      closeM(m.id.replace('modal-', ''));
    });
    return;
  }

  // / — focus search (only when not typing in a field)
  if (e.key === '/' && !inInput) {
    e.preventDefault();
    var si = document.getElementById('sinput');
    si.focus(); si.select();
    return;
  }

  // Arrow keys — navigate cards in detail modal
  if ((e.key === 'ArrowRight' || e.key === 'ArrowLeft') && document.getElementById('modal-detail').classList.contains('open')) {
    e.preventDefault();
    var cards = Array.from(document.querySelectorAll('.pc'));
    var idx = _currentPC ? cards.indexOf(_currentPC) : -1;
    if (idx === -1) return;
    var next = e.key === 'ArrowRight' ? cards[idx + 1] : cards[idx - 1];
    if (next) openDetail(next);
    return;
  }
});

function doS(q) {
  var r = document.getElementById('sres');
  if (!q) { hideS(); return; }
  if (!sdata) {
    r.style.display = 'block';
    r.innerHTML = [1,2,3].map(function() {
      return '<div class="sk-item"><div class="sk-thumb"></div><div class="sk-lines"><div class="sk-l" style="width:68%"></div><div class="sk-l" style="width:40%"></div></div></div>';
    }).join('');
    document.getElementById('sinput').setAttribute('aria-expanded','true');
    return;
  }
  var ql = q.toLowerCase();
  var f = sdata.filter(function(p) {
    var m = (cfilter === 'all' || p.category === cfilter);
    return m && (p.name.toLowerCase().indexOf(ql) > -1 || (p.id && p.id.toLowerCase().indexOf(ql) > -1) || (p.formula && p.formula.toLowerCase().indexOf(ql) > -1));
  });
  var sv = document.getElementById('ssort').value;
  if (sv === 'az') f.sort(function(a,b) { return a.name.localeCompare(b.name); });
  else if (sv === 'haccp') f.sort(function(a,b) { return (b.haccp ? 1:0) - (a.haccp ? 1:0); });
  else if (sv === 'eco') f.sort(function(a,b) {
    var ae = a.name.toLowerCase().indexOf('eco') > -1 ? 1:0;
    var be = b.name.toLowerCase().indexOf('eco') > -1 ? 1:0;
    return be - ae;
  });
  f = f.slice(0, 8);
  if (!f.length) {
    var qs = q.replace(/</g,'&lt;').replace(/>/g,'&gt;');
    r.style.display = 'block';
    r.innerHTML = '<div class="sri-empty"><svg width="40" height="40" viewBox="0 0 40 40" fill="none"><circle cx="18" cy="18" r="10" stroke="#7A94A8" stroke-width="1.5"/><path d="M26 26l6 6" stroke="#7A94A8" stroke-width="1.5" stroke-linecap="round"/><path d="M13 18h10" stroke="#C9A96E" stroke-width="1.5" stroke-linecap="round" opacity=".5"/></svg><p>«<span>' + qs + '</span>» üzrə nəticə tapılmadı.<br>Axtarış ifadəsini dəyişin.</p></div>';
    document.getElementById('sinput').setAttribute('aria-expanded','true');
    return;
  }
  var h = f.map(function(p) {
    var ig = p.image ? ('<img class="sri-img" src="'+p.image+'" alt="'+(p.name||'Məhsul').replace(/"/g,'&quot;')+'" onerror="this.style.display=\'none\'">') : '';

    return '<div class="sri" onclick="goS(\''+p.category+'\')">'+ig+'<div><div class="sri-name">'+p.name+hb+'</div><div class="sri-cat">'+(CATS[p.category]||p.category)+'</div>'+(p.id?'<div class="sri-id">'+p.id+'</div>':'')+' </div></div>';
  }).join('');
  r.style.display = 'block';
  r.innerHTML = h;
  document.getElementById('sinput').setAttribute('aria-expanded','true');
}

// ── Müqayisə (Compare) ───────────────────────────────────────────────────────
var cmpList = [];

function toggleCmp(btn) {
  var pc = btn.closest('.pc');
  if (!pc) return;
  var name = pc.querySelector('.pc-name') ? pc.querySelector('.pc-name').textContent.trim() : '';
  var code = pc.querySelector('.pc-code') ? pc.querySelector('.pc-code').textContent.trim() : '';
  var imgEl = pc.querySelector('.pc-img');
  var img = imgEl ? imgEl.src : '';
  var specs = Array.from(pc.querySelectorAll('.pc-ul li')).map(function(li) { return li.textContent.trim().replace(/^[\u2014\-]\s*/, ''); });
  var pkg = pc.querySelector('.pc-pkg') ? pc.querySelector('.pc-pkg').textContent.replace('📦 ','').trim() : '—';
  var del = pc.querySelector('.pc-del') ? pc.querySelector('.pc-del').textContent.trim() : '—';
  var section = pc.closest('section[id]');
  var cat = section ? (CATS[section.id] || section.id) : '—';
  var bdgEl = pc.querySelector('.pc-badges');
  var badges = bdgEl ? Array.from(bdgEl.querySelectorAll('.bdg')).map(function(b){return b.textContent;}) : [];

  var idx = cmpList.findIndex(function(x) { return x.name === name; });
  if (idx > -1) {
    cmpList.splice(idx, 1);
    btn.classList.remove('in-cmp');
  } else {
    if (cmpList.length >= 3) { toast('Maksimum 3 məhsul müqayisə edilə bilər.', 'error'); return; }
    cmpList.push({ name: name, code: code, img: img, specs: specs, pkg: pkg, del: del, cat: cat, badges: badges });
    btn.classList.add('in-cmp');
  }
  renderCmpBar();
}

function renderCmpBar() {
  var bar = document.getElementById('cmp-bar');
  var slotsEl = document.getElementById('cmp-slots');
  var countEl = document.getElementById('cmp-count');
  if (!cmpList.length) { bar.classList.remove('open'); return; }
  bar.classList.add('open');
  countEl.textContent = cmpList.length + ' / 3';
  slotsEl.innerHTML = cmpList.map(function(p, i) {
    return '<div class="cmp-slot filled"><span style="overflow:hidden;text-overflow:ellipsis;white-space:nowrap;max-width:120px">' + p.name.substring(0,22) + (p.name.length>22?'…':'') + '</span>' +
      '<button class="cmp-slot-rm" onclick="removeCmpAt(' + i + ')" aria-label="Sil">×</button></div>';
  }).concat(Array(3 - cmpList.length).fill('<div class="cmp-slot">Məhsul seçin...</div>')).join('');
}

function removeCmpAt(i) {
  var removed = cmpList.splice(i, 1)[0];
  document.querySelectorAll('.pc-cmp').forEach(function(btn) {
    var pc = btn.closest('.pc');
    if (pc && pc.querySelector('.pc-name') && pc.querySelector('.pc-name').textContent.trim() === removed.name) {
      btn.classList.remove('in-cmp');
    }
  });
  renderCmpBar();
}

  document.getElementById('cmp-bar').classList.remove('open');

// ── Axtarış tarixi ────────────────────────────────────────────────────────────
var HISTORY_KEY = 'mcg_search_history';
function getHistory() {
  try { return JSON.parse(localStorage.getItem(HISTORY_KEY) || '[]'); } catch(e) { return []; }
}
function saveHistory(q) {
  if (!q || q.length < 2) return;
  var h = getHistory().filter(function(x) { return x !== q; });
  h.unshift(q);
  localStorage.setItem(HISTORY_KEY, JSON.stringify(h.slice(0, 6)));
}
function showHistory() {
  var h = getHistory();
  if (!h.length) return;
  var r = document.getElementById('sres');
  r.style.display = 'block';
  r.innerHTML = '<div style="padding:8px 14px 4px;font-size:8px;color:var(--muted);letter-spacing:.14em;text-transform:uppercase;display:flex;justify-content:space-between;align-items:center">Son axtarışlar <span onclick="clearHistory()" style="cursor:pointer;color:var(--gold);font-size:8px">Sil</span></div>' +
    h.map(function(q) {
      return '<div class="sri" onclick="applyHistory(\'' + q.replace(/'/g,"\\'").replace(/</g,'&lt;') + '\')" style="padding:9px 14px;gap:10px"><span style="color:var(--muted);font-size:13px">🕐</span><span class="sri-name">' + q.replace(/</g,'&lt;') + '</span></div>';
    }).join('');
  document.getElementById('sinput').setAttribute('aria-expanded', 'true');
}
function applyHistory(q) {
  var si = document.getElementById('sinput');
  si.value = q; doS(q);
}
function clearHistory() {
  localStorage.removeItem(HISTORY_KEY);
  hideS();
}

document.getElementById('sinput').addEventListener('focus', function() {

    updateBasketCount();
  }
}

function populateQuoteBasket() {
  var box = document.getElementById('qbasket-box');
  var list = document.getElementById('qbasket-list');
  if (!quoteBasket.length) { box.style.display = 'none'; return; }
  box.style.display = 'block';
  list.innerHTML = quoteBasket.map(function(item) {
    return '<li><span style="color:var(--white);flex:1">' + item.name.replace(/</g,'&lt;') + '</span>' +
      (item.code ? '<span class="qb-code">' + item.code.replace(/</g,'&lt;') + '</span>' : '') + '</li>';
  }).join('');
  var cats = quoteBasket.map(function(x) { return x.category; }).filter(Boolean);
  document.querySelectorAll('#qform input[name="cats"]').forEach(function(cb) {
    if (cats.indexOf(cb.value) > -1) {
      cb.checked = true;
      cb.closest('.fcat-item').classList.add('checked');
    }
  });
  var notesEl = document.querySelector('#qform textarea[name="notes"]');
  if (notesEl && !notesEl.value) {
    notesEl.value = 'Seçilmiş məhsullar:\n' + quoteBasket.map(function(x, i) {
      return (i + 1) + '. ' + x.name + (x.code ? ' (' + x.code + ')' : '');
    }).join('\n');
  }
}

// ── Product detail modal ──────────────────────────────────────────────────────
function openDetail(pc) {
  _currentPC = pc;
  var code = pc.querySelector('.pc-code') ? pc.querySelector('.pc-code').textContent.trim() : '';
  var name = pc.querySelector('.pc-name') ? pc.querySelector('.pc-name').textContent.trim() : '';
  var imgEl = pc.querySelector('.pc-img');
  var img = imgEl ? imgEl.src : '';
  var imgAlt = imgEl ? (imgEl.getAttribute('alt') || '') : '';
  var specs = Array.from(pc.querySelectorAll('.pc-ul li')).map(function(li) {
    return li.textContent.trim().replace(/^[\u2014\-]\s*/, '');
  });
  var pkg = pc.querySelector('.pc-pkg') ? pc.querySelector('.pc-pkg').textContent.trim() : '';
  var del = pc.querySelector('.pc-del') ? pc.querySelector('.pc-del').textContent.trim() : '';
  var bdgEl = pc.querySelector('.pc-badges');
  var bdgHtml = bdgEl ? '<div style="display:flex;gap:5px;flex-wrap:wrap;margin-bottom:12px">' + bdgEl.innerHTML + '</div>' : '';
  var imgHtml = img
    ? '<img class="dm-img" src="' + img + '" alt="' + imgAlt.replace(/"/g,'&quot;') + '" loading="lazy" onerror="this.parentElement.innerHTML=\'<div class=dm-noimg>&#129529;</div>\'">'
    : '<div class="dm-noimg">🧴</div>';
  var specsHtml = specs.length
    ? '<ul class="dm-specs">' + specs.map(function(s) { return '<li>' + s.replace(/</g,'&lt;').replace(/>/g,'&gt;') + '</li>'; }).join('') + '</ul>'
    : '';
  var rowsHtml = '';
  if (pkg) rowsHtml += '<div class="dm-row"><span class="dm-label">Qablaşdırma</span><span class="dm-val">' + pkg.replace('📦 ','') + '</span></div>';
  if (del) rowsHtml += '<div class="dm-row"><span class="dm-label">Çatdırılma</span><span class="dm-val">' + del + '</span></div>';
  var section = pc.closest('section[id]');
  var cat = section ? section.id : '';
  if (cat && CATS[cat]) rowsHtml += '<div class="dm-row"><span class="dm-label">Kateqoriya</span><span class="dm-val">' + CATS[cat] + '</span></div>';
  document.getElementById('dm-body').innerHTML =
    '<div class="dm-side">' + imgHtml + '</div>' +
    '<div class="dm-info">' +
      bdgHtml +
      (code ? '<div class="dm-code">' + code.replace(/</g,'&lt;') + '</div>' : '') +
      '<div class="dm-name">' + name.replace(/</g,'&lt;') + '</div>' +
      specsHtml +
      (rowsHtml ? '<div style="margin-top:14px">' + rowsHtml + '</div>' : '') +
    '</div>';
  var addBtn = document.getElementById('dm-add-btn');
  var inBasket = quoteBasket.some(function(x) { return x.name === name; });
  if (inBasket) { addBtn.textContent = '✓ Sorğuda var'; addBtn.classList.add('dm-added'); }
  else { addBtn.textContent = '+ Sorğuya Əlavə Et'; addBtn.classList.remove('dm-added'); }
  var title = name.length > 35 ? name.substring(0, 32) + '…' : name;
  document.getElementById('modal-detail-title').textContent = title;
  openM('detail');
}

function addCurrentToQuote() {
  if (!_currentPC) return;
  var name = _currentPC.querySelector('.pc-name') ? _currentPC.querySelector('.pc-name').textContent.trim() : '';
  var code = _currentPC.querySelector('.pc-code') ? _currentPC.querySelector('.pc-code').textContent.trim() : '';
  var section = _currentPC.closest('section[id]');
  var cat = section ? section.id : '';
  var idx = quoteBasket.findIndex(function(x) { return x.name === name; });
  var addBtn = document.getElementById('dm-add-btn');
  var cardBtn = _currentPC.querySelector('.pc-add');
  if (idx === -1) {
    quoteBasket.push({ name: name, code: code, category: cat });
    addBtn.textContent = '✓ Sorğuda var'; addBtn.classList.add('dm-added');
    if (cardBtn) { cardBtn.classList.add('added'); cardBtn.textContent = '✓'; }
    updateBasketCount();
    toast('Sorğuya əlavə edildi', 'success');
  } else {
    quoteBasket.splice(idx, 1);
    addBtn.textContent = '+ Sorğuya Əlavə Et'; addBtn.classList.remove('dm-added');
    if (cardBtn) { cardBtn.classList.remove('added'); cardBtn.textContent = '+'; }
    updateBasketCount();
    toast('Sorğudan çıxarıldı', '');
  }
}

// ── Section TOC indicator ──────────────────────────────────────────────────
(function() {
  var toc = document.getElementById('sec-toc');
  if (!toc) return;
  var dots = Array.from(toc.querySelectorAll('.toc-dot'));
  var secs = dots.map(function(d) {
    return document.getElementById(d.dataset.sec);
  });

  var tocSections = ['kimyevi','camasirxana','metbex','amenity','istehlak','housekeeping','hovuz'];
  var firstSecTop = 0;
  var firstSec = document.getElementById(tocSections[0]);
  if (firstSec) firstSecTop = firstSec.getBoundingClientRect().top + window.scrollY - 200;

  function updateToc() {
    var scrollY = window.scrollY;
    var winH = window.innerHeight;

    // Show/hide TOC: show only when first catalog section is near view
    var inCatalog = scrollY >= firstSecTop;
    toc.classList.toggle('show', inCatalog);

    // Find active section
    var active = -1;
    secs.forEach(function(sec, i) {
      if (!sec) return;
      var top = sec.getBoundingClientRect().top + scrollY;
      if (scrollY >= top - winH * 0.4) active = i;
    });

    dots.forEach(function(d, i) {
      d.classList.toggle('active', i === active);
    });
  }

  window.addEventListener('scroll', updateToc, { passive: true });
  updateToc();
})();

// ── Print / PDF catalog ────────────────────────────────────────────────────
function printCatalog() {
  // Update print date
  var meta = document.getElementById('ph-meta');
  if (meta) {
    var now = new Date();
    var months = ['Yanvar','Fevral','Mart','Aprel','May','İyun','İyul','Avqust','Sentyabr','Oktyabr','Noyabr','Dekabr'];
    meta.textContent = 'Çap tarixi: ' + now.getDate() + ' ' + months[now.getMonth()] + ' ' + now.getFullYear();
  }
  // Count visible cards for user feedback
  var visible = document.querySelectorAll('.pc:not(.hidden)').length;
  var total = document.querySelectorAll('.pc').length;
  var msg = visible < total
    ? 'Hal-hazırda ' + visible + ' məhsul süzülüb. Yalnız bunlar çap ediləcək. Davam edirsiniz?'
    : null;
  if (msg && !confirm(msg)) return;
  window.print();
}
